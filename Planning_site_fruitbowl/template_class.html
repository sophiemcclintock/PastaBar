Brief

This python program is a pasta bar ordering system. It is designed so that users will be able to easily place an online order. There will be an option to add to their order, delete from their order as well as editing it along the way. We are using the menu from the local Wellington pasta resturant, 1154.

def get_integer(m, _min=0, _max=10):
    # assume it is wrong
    have_correct_value = False
    while have_correct_value is False:
        try:
            my_integer = int(input(m))
        except ValueError:
            print("You have not entered an integer")
            # immediate jumpback to top of loop
            continue
        # we have an integer
        if my_integer < _min:
            output = "Your value is too small, please re-enter"
            print(output)
            continue
        elif my_integer >= _max:
            output = "Your value is too big, please re-enter"
            print(output)
            continue
        # have the correct value so end loop
        return my_integer

def customer_details(c):
    name = get_string("Can you please enter your name -> ", 0, 50)
    phone_number = get_string("Can you please enter your phone number -> ", 0, 15)
    print("-"*80)
    options = ["D : Delivery", "P : Pickup"]
    for x in options:
        print(x)
    user_choice = get_string("Please enter your choice -> ", 0, 1)
    if user_choice == "D":
        delivery = True
        while delivery is True:
            number = get_string("You have selected Delivery. Can you please enter your address. Number -> ", 0, 5)
            street = get_string("Street -> ", 0, 50)
            suburb = get_string("Suburb -> ", 0, 50)
            city = get_string("City -> ", 0, 50)
            postcode = get_string("Postcode -> ", 0, 4)
            address = "{} {}, {} {} {}".format(number, street, suburb, city, postcode)
            output = get_string("Your address is {}. Is this correct? (Please enter Y or N) -> ".format(address), 0, 1)
            if output == "Y":
                delivery = False
            elif output == "N":
                delivery = True
        output = "Address: {}, Phone Number: {}, Name: {}".format(address, phone_number, name)
        c.extend([name, phone_number, address])
        print(output)
    elif user_choice == "P":
        print("You have selected Pickup")
        output = "Phone Number: {}, Name: {}".format(phone_number, name)
        c.extend([name, phone_number])
        print(output)

def finalising_order(o, c):
    review_order(o)
    confirm = get_string("Is this order correct? Please enter Yes (Y) or No (No) -> ", 0, 1)
    correcting_order = True
    while correcting_order is True:
        if confirm == "Y":
            customer_details(c)
            print("Your order has been finalised")
            o.clear()
            c.clear()
            order_or_quit = order_again()
            return order_or_quit
            correcting_order = False
        elif confirm == "N":
            correcting_order = False
            return True
        else:
            print("Unrecognised entry")
    
def order_again():
    output = get_string("Would you like to place another order? Please enter Yes (Y) or No (No) -> ", 0, 1)
    if output == "Y":
        return True
    elif output == "N":
        print("Thank you for ordering your meal. The program will terminate")
        return False
    else:
        print("Unrecognised entry")
